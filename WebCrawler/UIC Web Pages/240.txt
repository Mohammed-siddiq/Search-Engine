<script>
	// The amount of time for an image to cycle (fading in an image takes 1 second)
	var cycleTime = 6000;
	
	// HTMLDivElement, corresponds to #featured_area
	var area = null;
	
	// HTMLCollections, correspond to dynamic elements
	var descriptions = null;
	var images = null;
	var thumbnailRows = null;
	var titles = null;
	
	// Related to animation
	var active = 0;
	var numItems = 0;
	
	// Related to clearing timeout
	var cycleTimeout = undefined;
	var fadeDownTimeout = undefined;
	var fadeUpTimeout = undefined;
	
	function cycle() {
		var next = (active+1)%numItems;
		
		fadeDown(active, 1.0);
		images[next].style.display = "block";
		fadeUp(next, 0.0);
		
		active = next;
		
		cycleTimeout = setTimeout(function() {
			cycle();
		}, cycleTime);
	}
	
	function fadeDown(inId, inValue) {
		if (inValue <= 0.0) {
			images[inId].style.display = "none";
			images[inId].style.opacity = 0.0;
			swap(inId);
		}
		else {
			images[inId].style.opacity = inValue;
			
			fadeDownTimeout = setTimeout(function() {
				fadeDown(inId, inValue-0.0167);
			}, 16);
		}
	}
	
	function fadeUp(inId, inValue) {
		if (inValue >= 1.0) images[inId].style.opacity = 1.0;
		else {
			images[inId].style.opacity = inValue;
			
			fadeUpTimeout = setTimeout(function() {
				fadeUp(inId, inValue+0.0167);
			}, 16);
		}
	}
	
	function initialize() {
		// This section is highly dependent on HTML structure
		if (!(area = document.getElementById("featured_area"))) return;
		
		descriptions = document.getElementById("featured_descriptions");
		images = document.getElementById("featured_images");
		thumbnailRows = document.getElementById("featured_thumbnails");
		titles = document.getElementById("featured_titles");
		
		if (!descriptions || !images || !thumbnailRows || !titles) {
			area.style.display = "none";
			return;
		}
		
		descriptions = descriptions.getElementsByTagName("div");
		images = images.getElementsByTagName("img");
		thumbnailRows = thumbnailRows.getElementsByTagName("tr");
		titles = titles.getElementsByTagName("h3");
		
		numItems = images.length;
		if (numItems != descriptions.length || numItems != thumbnailRows.length || numItems != titles.length) {
			area.style.display = "none";
			return;
		}
		
		// Initialize CSS of dynamic elements
		for (var i = 1; i < numItems; i++) {
			descriptions[i].style.display = "none";
			images[i].style.display = "none";
			images[i].style.opacity = 0;
			titles[i].style.display = "none";
		}
		thumbnailRows[0].style.background = "#7070e0";
		thumbnailRows[0].style.color = "#f0f0f0";
		
		// Add event handlers to each thumbnail row
		var addHandeler = function(inId) {
			if (thumbnailRows[inId].addEventListener) thumbnailRows[inId].addEventListener("mousedown", function() { resetActive(inId); });
			else thumbnailRows[inId].attachEvent("onmousedown", function() { resetActive(inId); });
		};
		for (var i = 0; i < numItems; i++) addHandeler(i);
		
		cycleTimeout = setTimeout(function() {
			cycle();
		}, cycleTime);
	}
	
	function swap(inId) {
		descriptions[inId].style.display = "none";
		titles[inId].style.display = "none";
		thumbnailRows[inId].style.background = "";
		thumbnailRows[inId].style.color = "";
		
		var next = (inId+1)%numItems;
		
		descriptions[next].style.display = "block";
		titles[next].style.display = "block";
		thumbnailRows[next].style.background = "#7070e0";
		thumbnailRows[next].style.color = "#f0f0f0";
	}
	
	function resetActive(inId) {
		clearTimeout(cycleTimeout);
		clearTimeout(fadeDownTimeout);
		clearTimeout(fadeUpTimeout);
		
		for (var i = 0; i < numItems; i++) {
			descriptions[i].style.display = "none";
			images[i].style.display = "none";
			images[i].style.opacity = 0;
			titles[i].style.display = "none";
			thumbnailRows[i].style.background = "";
			thumbnailRows[i].style.color = "";
		}
		
		descriptions[inId].style.display = "block";
		images[inId].style.display = "block";
		images[inId].style.opacity = 1;
		titles[inId].style.display = "block";
		thumbnailRows[inId].style.background = "#7070e0";
		thumbnailRows[inId].style.color = "#f0f0f0";
		
		active = inId;
		cycleTimeout = setTimeout(function() {
			cycle();
		}, cycleTime);
	}
	
	initialize();
</script>
