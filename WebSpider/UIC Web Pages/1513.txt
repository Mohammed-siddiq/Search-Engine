		var num_words = words.length;
		word_lens = [];

		for (var j = 0; j < words.length; ++j) {
			word_lens.push(words[j].length);
		}

		var display_amount = 0;
		var total = 0;
		var index = 0;
		while (total < max_chars) {
			display_amount++;
			total += word_lens[index];
			index++;
		}

		words = words.slice(0, display_amount);
		new_title = words.join(' ');

		if (display_amount < num_words) {
			new_title = new_title + '...';
		}
		title.innerHTML = new_title;
	}

        shorten_excerpt();
	return;
}
